var myCal; var hour = 0; var minute = 0; var unixTime; function setupCalendars() { // Embedded Calendar myCal = Calendar.setup( { parentElement: 'embeddedCalendar' } ) } function calculateUnixTime() { var tempDate = new Date(myCal.date); var year = tempDate.getFullYear(); var day = tempDate.getDate(); var month = tempDate.getMonth() + 1; var newDate = new Date(year+'-'+month+'-'+day); unixTime = newDate.getTime() / 1000 +hour +minute; console.log(unixTime); var unixinput = document.getElementById('unixtime'); unixinput.value = unixTime; } function onHourChanged(value) { hour = parseInt(value) * 3600 calculateUnixTime(); } function onMinuteChanged(value) { minute = parseInt(value) * 60; calculateUnixTime(); } Event.observe(window, 'load', function() { setupCalendars() }) setLocation(0,0); setAllowEdit(true);
